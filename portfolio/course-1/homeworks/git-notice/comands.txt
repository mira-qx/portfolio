git init - создает новый репозиторий
git add - добавляет файлы в индекс
git status - показывает статус файлов
git commit -m "Сообщение коммита" - Сохранение изменений из индекса в репозиторий с сообщением.
git config --global user.name "[name]" - Установка имени пользователя для всех репозиториев.
git config --global user.email "[email]" - Установка электронной почты пользователя для всех репозиториев.
